```java
Scanner scn = new Scanner(System.in); System.out.println("0~100사이의 정수를 입력하세요");
int bar = scn.nextInt();
String result;
//90 이상 A
if(bar >= 90&&100>=bar){
result="A";
}
//80이상 B
else if(bar >= 80){
result="B";
}
//70이상 C
else if(bar >= 70){
result="C";
}
//60이상 D
else if(bar >= 60){
result="F";
}
//60미만 F
else if(bar < 60){
result="F";
}
//100초과 0미만 ->"입력값 오류"
else{
result="입력값 오류";
}
System.out.println(result);
```
